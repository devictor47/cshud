# 🔥 1. Don’t rebuild icons every update (YOU WILL REGRET IT)

Right now `createIcon()` + `appendChild()` is fine for testing, but with live updates:

* 10 players × ~10 updates/sec = 💀
* tons of DOM churn
* layout + paint spikes

### ✅ Fix: cache AND reuse DOM nodes per player

Instead of:

```js
container.innerHTML = "";
container.appendChild(icon);
```

Do:

```js
function setWeapon(container, iconName) {
    if (!iconName) {
        container.innerHTML = "";
        return;
    }

    let existing = container.firstChild;

    if (existing && existing.dataset.icon === iconName) {
        return; // already correct → no DOM change
    }

    container.innerHTML = "";

    createIcon(iconName).then(icon => {
        icon.dataset.icon = iconName;
        container.appendChild(icon);
    });
}
```

# ⚡ 3. Fast update function (this is your core loop)

```js
function updatePlayer(p) {
    const ui = playerBindings.get(p.id);
    if (!ui) return;

    // TEXT
    ui.name.textContent = p.name;
    ui.kills.childNodes[1].nodeValue = p.kills;
    ui.deaths.childNodes[1].nodeValue = p.deaths;
    ui.money.textContent = p.money;

    // HP
    ui.hpText.textContent = p.hp;
    ui.hpBar.style.setProperty("--hp", p.hp);

    // WEAPONS
    const primary = weaponData[p.primary]?.icon;
    const secondary = weaponData[p.secondary]?.icon;

    setWeapon(ui.primary, primary);
    setWeapon(ui.secondary, secondary);

    // ITEMS (grenades etc)
    updateItems(ui.items, p.items);
}
```

# 💣 4. Items diffing (don’t clear every time)

Instead of:

```js
items.innerHTML = "";
```

Do a minimal diff:

```js
function updateItems(container, items) {
    const existing = new Set(
        [...container.children].map(el => el.dataset.type)
    );

    const next = new Set(items);

    // remove old
    container.querySelectorAll("img").forEach(el => {
        if (!next.has(el.dataset.type)) {
            el.remove();
        }
    });

    // add new
    for (const type of items) {
        if (existing.has(type)) continue;

        const img = document.createElement("img");
        img.dataset.type = type;

        switch (type) {
            case "he":
                img.src = "overviews/hud/he.png";
                img.classList.add("he");
                break;
            case "flash":
                img.src = "overviews/hud/fb.png";
                img.classList.add("fb");
                break;
            case "smoke":
                img.src = "overviews/hud/sg.png";
                img.classList.add("sg");
                break;
            case "c4":
                img.src = "overviews/hud/c4.png";
                img.classList.add("c4");
                break;
            case "kit":
                img.src = "overviews/hud/kit.png";
                img.classList.add("kit");
                break;
        }

        container.appendChild(img);
    }
}
```

# 🧭 5. Active weapon (you mentioned this earlier)

You now have inventory, but not active weapon.

### Best UX:

* Primary slot shows **active if primary is active**
* Otherwise dim it

```css
.icon.inactive {
    opacity: 0.35;
    filter: none;
}
```

```js
function updateActive(ui, activeId) {
    const activeIcon = weaponData[activeId]?.icon;

    [ui.primary, ui.secondary].forEach(container => {
        const el = container.firstChild;
        if (!el) return;

        if (el.dataset.icon === activeIcon) {
            el.classList.remove("inactive");
        } else {
            el.classList.add("inactive");
        }
    });
}
```
